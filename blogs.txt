KMP-booth's algorithm 
https://en.wikipedia.org/wiki/Knuth%E2%80%93Morris%E2%80%93Pratt_algorithm#Variants
int rot(string s){
    s+=s;
    int n=s.length();
    int f[n];
    rep(i,0,n)f[i]=-1;
    int k=0;
    rep(j,1,n){
        int sj=s[j];
        int i=f[j-k-1];
        while(i!=-1 && sj!=s[k+i+1]){
            if(sj<s[k+i+1])k=j-i-1;
            i=f[i];
        }
        if(sj!=s[k+i+1]){
            if(sj<s[k])k=j;
            f[j-k]=-1;
        }
        else{
            f[j-k]=i+1;
        }

    }
    return k;

}
